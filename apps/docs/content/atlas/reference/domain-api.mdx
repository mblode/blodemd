title: Domain API reference
---

Use the Vercel Domains API to automate domain management for tenants.

## Add domain

```ts title="add-domain.ts"
import { Vercel } from "@vercel/sdk";

const vercel = new Vercel({
  bearerToken: "<YOUR_BEARER_TOKEN_HERE>",
});

async function run() {
  const result = await vercel.domains.createOrTransferDomain({
    teamId: "team_1a2b3c4d5e6f7g8h9i0j1k2l",
    slug: "my-team-url-slug",
    requestBody: {
      name: "example.com",
      method: "add",
      token: "fdhfr820ad#@FAdlj$$",
    },
  });

  console.log(result);
}

run();
```

```ts title="get-domain-status.ts"
const vercel = new Vercel({

    slug: "my-team-url-slug",
  });

  console.log(result);
}

run();
```

## Verify project domain ownership

```ts title="verify-domain.ts"
import { VercelCore as Vercel } from "@vercel/sdk/core.js";
import { projectsGetProjectDomain } from "@vercel/sdk/funcs/projectsGetProjectDomain.js";
import { projectsVerifyProjectDomain } from "@vercel/sdk/funcs/projectsVerifyProjectDomain.js";

const vercel = new Vercel({
  bearerToken: process.env.VERCEL_TOKEN,
});

const domain = "example.com";

async function run() {
  const [domainResponse, verifyResponse] = await Promise.all([
    projectsGetProjectDomain(vercel, {
      idOrName: "my-multi-tenant-app",
      teamId: "team_1a2b3c4d5e6f7g8h9i0j1k2l",
      domain,
    }),
    projectsVerifyProjectDomain(vercel, {
      idOrName: "my-multi-tenant-app",
      teamId: "team_1a2b3c4d5e6f7g8h9i0j1k2l",
      domain,
    }),
  ]);

  const { value: result } = verifyResponse;
  console.log({ domainResponse, verified: result?.verified ?? false });
}

run();
```

Use `vercel.domains.getDomainConfig()` when you need raw DNS configuration details. Use the project-scoped verification APIs above when you need to confirm tenant ownership on a specific project.

## Remove domain

```ts title="remove-domain.ts"
import { Vercel } from "@vercel/sdk";

const vercel = new Vercel({
  bearerToken: "<YOUR_BEARER_TOKEN_HERE>",
});

async function run() {
  const result = await vercel.domains.deleteDomain({
    domain: "example.com",
    teamId: "team_1a2b3c4d5e6f7g8h9i0j1k2l",
    slug: "my-team-url-slug",
  });

  console.log(result);
}

run();
```

## List domains

```ts title="list-domains.ts"
import { Vercel } from "@vercel/sdk";

const vercel = new Vercel({
  bearerToken: "<YOUR_BEARER_TOKEN_HERE>",
});

async function run() {
  const result = await vercel.domains.getDomains({
    limit: 20,
    since: 1609499532000,
    until: 1612264332000,
    teamId: "team_1a2b3c4d5e6f7g8h9i0j1k2l",
    slug: "my-team-url-slug",
  });

  console.log(result);
}

run();
```

## Error codes

| Code                    | Description                       | Solution                         |
| ----------------------- | --------------------------------- | -------------------------------- |
| `domain_already_in_use` | Domain is used by another project | Verify ownership with TXT record |
| `invalid_domain`        | Domain format is invalid          | Check domain spelling and format |
| `forbidden`             | Insufficient permissions          | Check API token permissions      |
| `rate_limit_exceeded`   | Too many requests                 | Retry with exponential backoff   |
